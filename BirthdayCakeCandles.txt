public class BirthdayCakeCandles {

    public static void main(String[] args) {

        int[] candle = {4, 4, 1, 3};

        // Step 1: Find the tallest candle
        int maxHeight = candle[0];
        for (int i = 1; i < candle.length; i++) {
            if (candle[i] > maxHeight) {
                maxHeight = candle[i];
            }
        }

        // Step 2: Count how many candles have the tallest height
        int count = 0;
        for (int i = 0; i < candle.length; i++) {
            if (candle[i] == maxHeight) {
                count++;
            }
        }

        // Output result
        System.out.println(count);
    }
}
